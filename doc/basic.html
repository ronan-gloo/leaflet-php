Basic Usage example

<?php 
// Creates a fresh map: the 'map' is the variable name and map identifier
$map = new Map('map');
// Set coordinates and Zoom
$map->setView([51.505, -0.09], 13);

// We now need a Tile Layer. Invoke Ovi 'terrain' tiles.
$tile = new Layer\Tile\Ovi(['style' => 'terrain']);
$tile->addTo($map);

// Make a marker and add to the map
$marker = new Layer\Marker([51.5, -0.09]);
$marker->addTo($map);

// Design a Circle
$circle = new Layer\Vector\Circle([51.508, -0.11], 500, [
  'color'       => 'red',
  'fillColor'   => '#f03',
  'fillOpacity' =>  0.5,
]);
$circle->addTo($map);

// Design a Simple Poygon
$polygon = new Layer\Vector\Polygon([[51.509, -0.08],[51.503, -0.06],[51.51, -0.047]]);
$polygon->addTo($map);

// Bind Popups on layers
$marker->bindPopup("<b>Hello world!</b><br>I am a popup.");
$circle->bindPopup("I am a circle.");
$polygon->bindPopup("I am a polygon.");

// Javascript var names are autogenerated, specify a name like this:
$marker->setRef('marker');

// If you need explicitly use the var later:
$marker->useRef()->openPopup();
