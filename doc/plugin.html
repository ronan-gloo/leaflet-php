Leaflet plugin example: Draw

<?php 
// Creates the plugin that extends the control base class
class DrawEdit extends Control\Control {
    // Plugin class name
    public function jsName()
    {
        return 'new L.Control.Draw';
    }
    // A Hook example : 
    public function addTo(Map $map)
    {
        parent::addTo($map);
        // Prevent json_encode string
        $layerVar = new JsVar('e.layer');
        //Respond to the different actions users can trigger
        $map->on('draw:created', function($c, $map) use($layerVar) {
            $c->args('e');
            $c->line($map->useRef()->addLayer($layerVar));
        });
    }
}

// create a map in the "map" div, set the view to a given place and zoom
$map = new Map('map', [
    'center' => [-37.77914, 175.30867],
    'zoom' => 13
]);

// add an OpenStreetMap tile layer
$layer = new Layer\Tile\Osm([
    'attribution' => '&copy; <a href="http://osm.org/copyright">OpenStreetMap</a> contributors'
]);
$layer->addTo($map);

// Initialize the FeatureGroup to store editable layers
$drawItems = new Layer\FeatureGroup();
$drawItems->addTo($map);

// Initialize the draw control and pass it the FeatureGroup of editable layers
$drawControl = new DrawEdit([
    'edit' => ['featureGroup' => $drawItems]
]);
$drawControl->addTo($map);
