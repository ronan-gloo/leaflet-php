Map events

<?php 
// Set option callback: don't wrap map
$layer = new Layer\Tile\Osm([
    'noWrap' => function($c){
        $c->line('return true;');
    }
]);

// Use the "onAdd" Control event
$control = new Control\Control();
$control->onAdd(function($c){
        $c->line("this._div = L.DomUtil.create('div', 'leaflet-control-layers');")
            ->line("this._div.innerHTML = '<h4>Control Legend</h4>';")
            ->line("return this._div;");
});

// Attache aa event to the map
$map = new Map('map');
$map->on('click', function($c){
        $c->args('e'); // Setup function argument
        $c->line('alert("click")'); // Add lineto function's body
});

$map
    ->fitWorld()
    ->addLayer($layer)
    ->addControl($control);
