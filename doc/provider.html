Providers Example

<?php 
// Setup some providers
$tile['Open Cycle Map']   = TileLayer::make('ocm',        ['style'    => 'landscape']);
$tile['Ersi Topo']        = TileLayer::make('esri',       ['style'    => 'world_topo_map']);
$tile['Mapbox']           = TileLayer::make('mapbox',     ['maxZoom'  => 8]);
$tile['Google Satellite'] = TileLayer::make('googlemap',  ['style'    => 'satellite']);
$tile['Ovi Hybrid']       = TileLayer::make('ovi',        ['style'    => 'hybrid']);
$tile['Open Street Map']  = TileLayer::make('osm');
$tile['MapQuest']         = TileLayer::make('mapquest');
$tile['Stamen']           = TileLayer::make('stamen');

// Example of Provider that requires key:
//$tile['Cloudmade'] = TileLayer::make('cloudmade', ['key' => 'youKeyHere']);

// Setup Extra layers
$extra['Open Weather Press']  = TileLayer::make('openweather',  ['style' => 'pressure_cntr']);
$extra['Google Labels']       = TileLayer::make('googlemap',    ['style' => 'label']);

// Your own provider
$extra['Open Sea Map'] = TileLayer::make('http://tiles.openseamap.org/seamark/{z}/{x}/{y}.png');

// Creates map and add first layer
$map = Map::make('map')->addLayer(reset($tile))->fitWorld();

// Creates control and add it to the map
Control::layers($tile, $extra)->addTo($map);
