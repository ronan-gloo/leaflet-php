Markers and Icons

<?php 
// Set map Layers
$layers = [
  new Layer\Tile\Ocm(),
  new Layer\Marker([39.61, -105.02]),
  new Layer\Marker([39.74, -104.99]),
  new Layer\Marker([39.73, -104.8]),
  new Layer\Marker([39.77, -105.23])
];

// Create a  icons
$icon = new Layer\Icon([
  'iconUrl'   => 'http://google-maps-icons.googlecode.com/files/tickmark1.png',
  'iconSize'  => 32
]);
$divIcon = new Layer\DivIcon([
  'iconSize'  => [64, 64],
  'html'      => 'Div Icon'
]);

// Add icon to the last marker using 'var = ref" javascript variable:
$layers[3]->useRef()->setIcon($icon);
$layers[4]->useRef()->setIcon($divIcon);

// Creates our map
$map = new Map('map', [
  'zoom'    => 10,
  'center'  => new LatLng(39.73, -104.99),
  'layers'  => $layers
]);
