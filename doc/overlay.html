Map standalone Overlays

<?php 
$map = new Map('map');
$map->setView([40.773941, -74.12544], 12);

$tile = new Layer\Tile\Stamen();
$tile->addTo($map);

// Creates a standalone popup
$popup = new Layer\Popup();
$popup
  ->setLatLng([40.773941, -74.12544])
  ->setContent('<p>Hello world!<br />This is a nice popup.</p>')
  ->openOn($map);

//Setup an imageoverlay
$iover = new Layer\ImageOverlay(
    'http://www.lib.utexas.edu/maps/historical/newark_nj_1922.jpg',
    [[40.712216, -74.22655], [40.773941, -74.12544]]
);
$iover->addTo($map);
