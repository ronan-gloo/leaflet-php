Objects and callbacks

<?php 
// Attache aa event to the map
$layer = new Layer\Tile\Ovi();

$map = new Map('map');
$map->addLayer($layer);
$map->locate(['setView' => true]);

// You can use string / or objets for the callback
$map->on('locationfound', function($c, $map){
    $coords = new JsVar('e.latlng');
    $c->args('e');
    $c->line((new Layer\Vector\Circle($coords, 400))->addTo($map));
    $c->line((new Layer\Marker($coords))->addTo($map));
});

$map->on('locationerror', function($c, $map){
    $c->args('e');
    $c->line('alert(e.message)');
});
