Map Controls

<?php 
$map = new Map('map');
$map->setView([51.505, -0.09], 3);

// Setup Attribution control
$attr = new Control\Attribution([
  'prefix'  => 'Leaflet PHP',
  'position'=> 'bottomleft'
]);
$attr
  ->addAttribution("Custom Attribution")
  ->addAttribution("Another Attribution")
  ->addTo($map);

// Setup Control Scale
$scale = new Control\Scale(['maxWidth' => 200]);
$scale->addTo($map);

// Set Layers Control
$tiles['Ovi'] = new Layer\Tile\Ovi(['style' => 'satellite']);
$tiles['Osm'] = new Layer\Tile\Osm();

// Display first tile by default
$tiles['Ovi']->addTo($map);

// Add an overlay layer
$overlay['Temp'] = new Layer\Tile\Openweather([
  'style'   => 'temp',
  'opacity' => 0.3
]);

// Set control Layer
$layers = new Control\Layers($tiles, $overlay, [
  'collapsed' => false
]);
$layers->addTo($map);
