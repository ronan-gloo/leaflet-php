Getting Started

<?php 

// First, you need to create a context to wrap javascript code.
// Context is like an environment, where code generation is wrapped
// Optional: Context options
$opts = [
  'refName'     => 'context1', // javascript vars basename
  'refIncrement'=> 'Aa', // javascript vars name auto-incrementation pattern
  'jsonFlags'   => JSON_UNESCAPED_SLASHES // json_encode flags
];

// Register a context
$contexts[] = new Context('leaflet', $opts);

// Unless you create a new context, all code above will be catched by $context
$map = new Map('map');
$map->locate(['setView' => true]);

$tile = new Layer\Tile\Ocm();
$tile->addTo($map);

// new Tile in another context
$contexts[] = new Context('leaflet2', ['refName' => 'context2']);
$tile2 = new Layer\Tile\Esri();

// Switch back to context 1 and add a new Layer
Context::switchTo('leaflet');
$tile3 = new Layer\Tile\Mapquest();

// Creates a new context for control layers.
// If no identifier is provided, the identifier is a uniqid() string
$contexts[] = new Context();
Control::layers([
  'Context 1a'  => $tile,
  'Context 2'   => $tile2,
  'Context 1b'  => $tile3,
])->addTo($map);

// Get The Javascript
$javascript = array_reduce($contexts, function(&$result, $context){
  return $result .= $context->getJs();
});
